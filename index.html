use std::sync::Arc;
use tokio::sync::RwLock;
/// Zero-copy tensor scheduler for GPU inference
pub struct TensorScheduler {
pool: Arc<MemoryPool>,
streams: Vec<CudaStream>,
queue: RwLock<VecDeque<InferTask>>,
}
impl TensorScheduler {
pub async fn dispatch(
&self,
input: Tensor<f16>,
) -> Result<Tensor<f16>> {
let stream = self.acquire_stream();
let buf = self.pool.alloc_zero_copy(
input.shape(),
stream.device(),
)?;
stream.launch_kernel(&buf, &input).await
}
}
